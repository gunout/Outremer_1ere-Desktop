# Outremer_1ere-Desktop
Outremer_1ere Desktop Application ( python ) ( Toutes les Radios 1ere Outremer ) 

# REQUIREMENTS 

    requests==2.31.0
    beautifulsoup4==4.12.2
    PyGObject==3.44.1
    python-vlc==3.0.18121


# INSTALL

    pip install -r requirements.txt


# VLC 

    sudo apt install vlc


# RUN 

    chmod +x la1ere.py
    python la1ere.py
